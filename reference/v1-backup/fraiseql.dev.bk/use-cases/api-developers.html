GraphQL That Scales 

# Complex nested queries? No problem.
@fraise_type
class Author:
    id: UUID
    name: str
    bio: str
    posts: list[Post]
    stats: AuthorStats

@fraise_type
class Post:
    id: UUID
    title: str
    content: str
    published_at: datetime
    tags: list[str]  # JSONB array
    comments: list[Comment]

@fraise_type
class AuthorStats:
    total_posts: int
    total_views: int
    avg_rating: float

# This complex query still executes as ONE SQL query
query = """
{
  authors(limit: 10) {
    name
    stats {
      total_posts
      avg_rating
    }
    posts(status: "published") {
      title
      tags
      comments(limit: 5) {
        text
        author_name
      }
    }
  }
}
"""

# Generated SQL uses LATERAL joins and JSONB aggregation
# Result: ~30ms for complex nested data vs 200ms+ with ORMs

Advanced Patterns 

# Cursor-based pagination with type safety
@fraise_type
class PageInfo:
    has_next_page: bool
    has_previous_page: bool
    start_cursor: str
    end_cursor: str

@fraise_type
class PostConnection:
    edges: list[PostEdge]
    page_info: PageInfo
    total_count: int

@fraiseql.query
async def posts(
    info,
    first: int = 20,
    after: str = None,
    filter: PostFilter = None
) -> PostConnection:
    """Relay-style pagination with filtering"""
    return await info.context.repo.paginate(
        "tv_post",
        first=first,
        after=after,
        where=filter.to_sql() if filter else None
    )

# Custom SQL when you need it
@fraiseql.query
async def trending_posts(info, hours: int = 24) -> list[Post]:
    """Complex analytics query with custom SQL"""
    sql = """
    SELECT data FROM v_post_analytics
    WHERE created_at > NOW() - INTERVAL '%s hours'
    ORDER BY
        (views * 0.3 + comments * 0.5 + shares * 0.2) DESC,
        created_at DESC
    LIMIT 20
    """
    return await info.context.repo.execute_sql(sql, [hours])

Integration Examples

FastAPI Integration

from fastapi import FastAPI
from fraiseql.fastapi import FraiseQLConfig, create_fraiseql_app

app = FastAPI()
config = FraiseQLConfig(database_url=DATABASE_URL)
fraiseql_app = create_fraiseql_app(
    types=[Author, Post, Comment],
    config=config,
)

app.mount("/graphql", fraiseql_app)

Async/Await Native

@fraiseql.query
async def search_posts(
    info,
    query: str,
    limit: int = 20
) -> list[Post]:
    # Full async support
    results = await info.context.repo.search(
        "v_post_search",
        query=query,
        limit=limit
    )

    # Post-process if needed
    return [enhance_post(p) for p in results]
