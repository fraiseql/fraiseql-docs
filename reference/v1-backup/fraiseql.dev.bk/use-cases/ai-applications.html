RAG Pipeline Example

# Complete RAG pipeline with FraiseQL
from langchain_community.document_loaders import WebBaseLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_openai import OpenAIEmbeddings, ChatOpenAI
from langchain.chains import RetrievalQA
from fraiseql.integrations.langchain import FraiseQLVectorStore

# 1. Load and process documents
loader = WebBaseLoader("https://fraiseql.dev/docs")
documents = loader.load()

text_splitter = RecursiveCharacterTextSplitter(
    chunk_size=1000, chunk_overlap=200
)
docs = text_splitter.split_documents(documents)

# 2. Store in FraiseQL with pgvector
embeddings = OpenAIEmbeddings()
vectorstore = FraiseQLVectorStore(
    graphql_url="http://localhost:8000/graphql",
    embeddings=embeddings,
    collection_name="documentation"
)
vectorstore.add_documents(docs)

# 3. Create RAG chain
llm = ChatOpenAI(temperature=0)
qa_chain = RetrievalQA.from_chain_type(
    llm=llm,
    chain_type="stuff",
    retriever=vectorstore.as_retriever(k=3),
    return_source_documents=True
)

# 4. Ask questions
result = qa_chain({"query": "How do I implement authentication?"})
print(result["result"])
