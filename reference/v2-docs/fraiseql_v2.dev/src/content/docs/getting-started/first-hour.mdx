In this tutorial, you'll build a **complete blog API** with users, posts, and comments. By the end, you'll have a working GraphQL API with CQRS patterns, relationships, and mutations.

## Prerequisites

Before starting, ensure you have:
- FraiseQL CLI installed (`fraiseql --version`)
- PostgreSQL running locally or accessible
- Your preferred schema language (Python 3.12+, Node.js 18+, or Go 1.21+)
- A terminal and text editor

## What We're Building

A blog API with:
- **Users**: Authors who write posts
- **Posts**: Blog entries with titles, content, and tags
- **Comments**: Threaded discussions on posts

```
┌─────────┐     ┌─────────┐     ┌───────────┐
│  User   │ ──→ │  Post   │ ──→ │  Comment  │
└─────────┘     └─────────┘     └───────────┘
## Part 1: Project Setup (5 minutes)

### Create the Project

```bash
mkdir blog-api && cd blog-api
fraiseql init
    ```

This creates:
```
blog-api/
├── fraiseql.toml      # Configuration
├── schema.*           # Type definitions
└── db/
    └── migrations/    # SQL migrations
    ```

### Configure the Database

Edit `fraiseql.toml`:

```toml
[project]
name = "blog-api"
version = "0.1.0"

[database]
type = "postgresql"
url = "postgresql://localhost/blog_api"

[server]
port = 8080
host = "0.0.0.0"
```

### Create the Database

```bash
createdb blog_api
    ```

        pk_user INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
        id UUID DEFAULT gen_random_uuid() UNIQUE NOT NULL,
        identifier TEXT UNIQUE NOT NULL,

        -- User data

        -- Metadata
        created_at TIMESTAMPTZ DEFAULT NOW(),

    -- Posts table
    CREATE TABLE IF NOT EXISTS tb_post (
        pk_post INTEGER GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
        id UUID DEFAULT gen_random_uuid() UNIQUE NOT NULL,
        identifier TEXT UNIQUE NOT NULL,

        -- Foreign key to author
        fk_user INTEGER NOT NULL REFERENCES tb_user(pk_user),

        -- Post data
        title TEXT NOT NULL,
        slug TEXT UNIQUE NOT NULL,
        content TEXT NOT NULL,
        excerpt TEXT,
        tags TEXT[] DEFAULT '{}',
        is_published BOOLEAN DEFAULT false,
        published_at TIMESTAMPTZ,
        view_count INTEGER DEFAULT 0,

### Read Views

Create `db/migrations/002_views.sql`:

<Tabs syncKey="db">
  <TabItem label="PostgreSQL">
    ```sql
    -- User view
    CREATE OR REPLACE VIEW v_user AS
    SELECT
        u.id,
        jsonb_build_object(
            'id', u.id::text,
            'identifier', u.identifier,
            'email', u.email,
            'name', u.name,
            'bio', u.bio,
            'avatar_url', u.avatar_url,
            'is_active', u.is_active,
            'created_at', u.created_at,
            'updated_at', u.updated_at
        ) AS data
    FROM tb_user u;

    -- Post view (with nested author)
    CREATE OR REPLACE VIEW v_post AS
    SELECT
        p.id,
        jsonb_build_object(
            'id', p.id::text,
            'identifier', p.identifier,
            'title', p.title,
            'slug', p.slug,
            'content', p.content,
            'excerpt', p.excerpt,
            'tags', p.tags,
            'is_published', p.is_published,
            'published_at', p.published_at,
            'view_count', p.view_count,
            'created_at', p.created_at,
            'updated_at', p.updated_at,
            'author', vu.data
        ) AS data
    FROM tb_post p
    JOIN tb_user u ON u.pk_user = p.fk_user
    JOIN v_user vu ON vu.id = u.id;

    -- Comment view (with nested post and author)
    CREATE OR REPLACE VIEW v_comment AS
    WITH RECURSIVE comment_tree AS (
        SELECT
            c.pk_comment,
            c.id,
            c.content,
            c.fk_post,
            c.fk_user,
            c.fk_parent_comment,
            c.created_at,
            c.updated_at,
            0 AS depth,
            ARRAY[c.pk_comment] AS path
        FROM tb_comment c
        WHERE c.fk_parent_comment IS NULL

        UNION ALL

        SELECT
            c.pk_comment,
            c.id,
            c.content,
            c.fk_post,
            c.fk_user,
            c.fk_parent_comment,
            c.created_at,
            c.updated_at,
            ct.depth + 1,
            ct.path || c.pk_comment
        FROM tb_comment c
        JOIN comment_tree ct ON ct.pk_comment = c.fk_parent_comment
        WHERE NOT c.pk_comment = ANY(ct.path)
    )
    SELECT
        c.id,
        jsonb_build_object(
            'id', c.id::text,
            'content', c.content,
            'depth', c.depth,
            'created_at', c.created_at,
            'updated_at', c.updated_at,
            'post', vp.data,
            'author', vu.data,
            'parent_comment', CASE
                WHEN pc.id IS NOT NULL THEN jsonb_build_object(
                    'id', pc.id::text,
                    'author', vu_pc.data
                )
                ELSE NULL
            END
        ) AS data
    FROM comment_tree c
    JOIN tb_post p ON p.pk_post = c.fk_post
    JOIN v_post vp ON vp.id = p.id
    JOIN tb_user u ON u.pk_user = c.fk_user
    JOIN v_user vu ON vu.id = u.id
    LEFT JOIN tb_comment pc ON pc.pk_comment = c.fk_parent_comment
    LEFT JOIN tb_user pu ON pu.pk_user = pc.fk_user
    LEFT JOIN v_user vu_pc ON vu_pc.id = pu.id;
    ```

            FOR JSON PATH, WITHOUT_ARRAY_WRAPPER

    CREATE OR REPLACE VIEW dbo.v_post
    WITH SCHEMABINDING AS
    SELECT
        p.id,
        (
            SELECT p.id, p.identifier, p.title, p.slug, p.content, p.excerpt, p.tags, p.is_published, p.published_at, p.view_count, p.created_at, p.updated_at, vu.data AS author

    JOIN dbo.tb_user u ON u.pk_user = p.fk_user
    JOIN dbo.v_user vu ON vu.id = u.id;

    -- Comment view (with nested post and author)
    CREATE OR REPLACE VIEW dbo.v_comment
    WITH SCHEMABINDING AS
    WITH comment_tree AS (
        SELECT
            c.pk_comment,
            c.id,
            c.content,
            c.fk_post,
            c.fk_user,
            c.fk_parent_comment,
            c.created_at,
            c.updated_at,
            0 AS depth,
            CAST(CONCAT('[', CAST(c.pk_comment AS VARCHAR(MAX)), ']') AS VARCHAR(MAX)) AS path
        FROM dbo.tb_comment c
        WHERE c.fk_parent_comment IS NULL

        UNION ALL

        SELECT
            c.pk_comment,
            c.id,
            c.content,
            c.fk_post,
            c.fk_user,
            c.fk_parent_comment,
            c.created_at,
            c.updated_at,
            ct.depth + 1,
            CONCAT(ct.path, ',', CAST(c.pk_comment AS VARCHAR(MAX)))
        FROM dbo.tb_comment c
        JOIN comment_tree ct ON ct.pk_comment = c.fk_parent_comment
        WHERE CHARINDEX(CAST(c.pk_comment AS VARCHAR(MAX)), ct.path) = 0
    )
    SELECT
        c.id,
        (
            SELECT c.id, c.content, c.depth, c.created_at, c.updated_at, vp.data AS post, vu.data AS author,
